 var queue = vm._hooks[hook];
    
      var step = function (index) {
        var hook = queue[index];
        if (index >= queue.length) {
          next(data);
        } else if (typeof hook === 'function') {
          if (hook.length === 2) {
            hook(data, function (result) {
              data = result;
              step(index + 1);
            });
          } else {
            var result = hook(data);
            data = result === undefined ? data : result;
            step(index + 1);
          }
        } else {
          step(index + 1);
        }
      };
